## 隐藏元素有三种方式
1. display:none
2. visibility: hidden
3. opacity:0

## display:none
```tsx
import { Modal } from "antd";
import React, { useEffect } from "react";

type Props = {};

const Demo = (props: Props) => {
	const btnRef = React.useRef<HTMLButtonElement>(null);
	const [open, setOpen] = React.useState(false);

	const handleClick = () => {
		setOpen(true);
	};

	useEffect(() => {
		const timer = setTimeout(() => {
			btnRef.current?.click();
		}, 4000);
		return () => {
			clearTimeout(timer);
		};
	}, []);

	return (
		<div>
			<button ref={btnRef} onClick={handleClick} style={{ display: "none" }}>
				Click Me
			</button>
			<Modal open={open} onCancel={() => setOpen(false)}>
				<p>Some contents...</p>
			</Modal>
		</div>
	);
};

export default Demo;

```
display: none 并不会让元素从 DOM（文档结构）中移除，它只是让元素不参与页面的布局与渲染，但依然存在于 DOM 树里。
