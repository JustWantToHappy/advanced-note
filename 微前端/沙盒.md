				if (sandboxRunning) {
					if (!rawWindow.hasOwnProperty(p)) {
						addPropsMapInSandbox.set(p, value);
					} else if (!modifiedPropsOriginalValueMapInSandbox.has(p)) {
						const originValue = rawWindow[p];
						modifiedPropsOriginalValueMapInSandbox.set(p, originValue);

				}

			}
