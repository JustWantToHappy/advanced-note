## 函数式泛型组件
### 一般使用方法
```tsx
export const JudgeChargeGame = () => {
	const [searchForm] = Form.useForm();
	const onSearch = () => {};

	return (
		<>
			<SearchForm<{ name: string }>
				searchForm={searchForm}
				options={[]}
				onSearch={onSearch}
			/>
		</>
	);
};


type Props<T> = {
	initialValues?: Partial<T>;
	searchForm: FormInstance<T>;
	onSearch: () => void;
	options: Array<{ label: string; value: string }>;
};

export const SearchForm: <T>(props: Props<T>) => JSX.Element = ({
	onSearch,
	options,
	searchForm,
	initialValues,
}) => {
	return (
		<header style={{ marginBottom: "2rem" }}>
			<Space size="large">
				<Form
					form={searchForm}
					name="GameFilter"
					layout="inline"
					style={{ textAlign: "left", height: 60 }}
					onFinish={onSearch}
					autoComplete="off"
				>
					<Space>
						<Form.Item
							label="搜索类型"
							name="key"
							rules={[
								{
									required: true,
									message: "搜索类型不能为空",
								},
							]}
						>
							<Select style={{ width: 200 }} options={options} />
						</Form.Item>
						<Form.Item
							label="搜索关键字"
							name="value"
							rules={[
								{
									required: true,
									message: "搜索关键词不能为空",
								},
							]}
						>
							<Input placeholder="请输入你要搜索的关键字"></Input>
						</Form.Item>
						<Form.Item>
							<Button ghost htmlType="submit" type="primary">
								点击搜索
							</Button>
						</Form.Item>
					</Space>
				</Form>
			</Space>
		</header>
	);
};

```
